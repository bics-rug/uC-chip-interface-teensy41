115 def process_packet(self, packet):
116 if packet.header() in self.__header:
117 if packet.header() == self.__header[0]:
118 if packet.config_header() == ConfigSubHeader.CONF_ACTIVE:
119 self.__status = 2
120 self.__status_timestamp = packet.time()
121 return
122 elif packet.config_header() == ConfigSubHeader.CONF_BYTE_ORDER:
123 self.__order = "MSBFIRST" if (packet.value() > 0) else "LSBFIRST"
124 self.__order_timestamp = packet.time()
125 return
126 elif packet.config_header() == ConfigSubHeader.CONF_WIDTH:
127 self.__number_of_bytes = packet.value()
128 self.__number_of_bytes_timestamp = packet.time()
129 return
130 elif packet.config_header() == ConfigSubHeader.CONF_SPEED_CLASS:
131 if packet.value() == 4:
132 self.__speed = 10000
133 elif packet.value() == 0:
134 self.__speed == 100000
135 elif packet.value() == 1:
136 self.__speed == 400000
137 elif packet.value() == 2:
138 self.__speed == 1000000
139 elif packet.value() == 3:
140 self.__speed == 3400000
141 else:
142 self.__speed = packet.value()
143 self.__speed_timestamp = packet.time()
144 return
145 elif packet.header() == self.__header[1]:
146 self.__data_to_chip.append((packet.read(),packet.device_address(),packet.register_address(),packet.value()))
147 self.__data_to_chip_times.append(packet.time())
148 return
149 elif packet.header() == self.__header[2]:
150 self.__data_from_chip.append((packet.read(),packet.device_address(),packet.register_address(),packet.value()))
151 self.__data_from_chip_times.append(packet.time())
152 return
153 self.__errors.append(str(packet))
154 self.__status = -1

156 def activate(self, speed=400000, order="LSBFIRST", number_of_bytes=1, time=0):
157 self.__api.update_state()
158 if self.__status >= 1:
159 logging.warning("I2C interface "+str(self.__header[0])+" is already activated or waiting activation, doing nothing")
160 else:
161 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_BYTE_ORDER, value= (1 if (order == "MSBFIRST") else 0),time = time))
162 if speed == 10000:
163 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_SPEED_CLASS, value=4,time = time))
164 elif speed == 100000:
165 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_SPEED_CLASS, value=0,time = time))
166 elif speed == 400000:
167 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_SPEED_CLASS, value=1,time = time))
168 elif speed == 1000000:
169 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_SPEED_CLASS, value=2,time = time))
170 elif speed == 3400000:
171 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_SPEED_CLASS, value=3,time = time))
172 else:
173 logging.warning("I2C only supports 10000,100000,400000,1000000,3400000 Hz, defaulting to 100000")
174 if number_of_bytes > 2:
175 logging.warning("I2C ony supports one or two bytes, defaulting to 1")
176 else:
177 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_WIDTH, value=number_of_bytes,time = time))
178 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_ACTIVE,time = time))
179 self.__status = 1

181 def send_write(self,device_address, register_address, word, time = 0):
182 self.__api.update_state()
183 if (time == 0 and self.__status == 2) or (time > 0 and self.__status >= 1):
184 self.__api.send_packet(DataI2CPacket(self.__header[1], device_address=device_address, register_address=register_address,read=0,value=word,time=time))
185 else:
186 logging.error("I2C interface "+str(self.__header[1])+" is not active, please activate first - word is not sent.")

192 else:
193 logging.error("I2C interface "+str(self.__header[2])+" is not active, please activate first - word is not sent.")
