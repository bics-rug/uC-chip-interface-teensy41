80 break;

84void exec_timed_commands(){
85 noInterrupts();
86 uint8_t attempt = 0;
87 if (exec_active) return;
88 exec_active = true;
89 uint32_t current_time = micros() - offset_time; // time does not move forward in interupts
90
91 packet_t instruction = copy_packet(&input_ring_buffer[input_ring_buffer_start]);
92
93 while (
94 (input_ring_buffer_start % INPUT_BUFFER_SIZE != input_ring_buffer_next_free) &&
95 instruction.data.exec_time <= current_time ) {
96 interrupts();
97 exec_instruction(&instruction,true);
98 noInterrupts();
99 input_ring_buffer_start = (input_ring_buffer_start + 1) % INPUT_BUFFER_SIZE;
100 instruction = copy_packet(&input_ring_buffer[input_ring_buffer_start]);
101 }
102 exec_active = false;
103 interrupts();
104}

106void set_time_offset(uint32_t offset){
107 if (offset){
108 offset_time = micros() + offset;
109 myTimer.begin(exec_timed_commands, EXEC_PRESISION);
110 //send confirmation
111 send_data32(IN_SET_TIME,offset,true);
112 }
113 else{
114 myTimer.end();
115 input_ring_buffer_start = input_ring_buffer_next_free;
116 //send confirmation
117 send_data32(IN_SET_TIME,offset,true);
118 offset_time = 0;
119 }

121}
