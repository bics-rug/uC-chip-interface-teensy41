31class uC_api:
32 """
33 the class uC_api exposes the full interface to the uC as an object,
34 all the interface like i2c, spi, pin and async can be found as exposed variables
35
36 Interfaces need to be activated before use, and the activation is ackloaged by the uC
37 you might need to time.sleep(0.01) after the activation for the python object to update register the state change
38
39 if you want to record data you need to use function start_experiment() to be abel to see the data send back
40 the maximum time of each experiment is 72min, please call stop experiment before that to start agian
41
42 after you are done call close_connection to sever the serial connection to the uC,
43 the recorded data in the python object remains and can be processed after
44 """
45 def __init__(self, serial_port_path, api_level=2):
46 """__init__ creates the uC interface object and establishes the connection to the uC on the given port
47
48 :param serial_port_path: the path of your system to the serial port, eg. on linux it might be /dev/ttyAMC0 or higher, on mac /dev/tty.usbmodem<XXXXX> on windows <COM port>
49 :type serial_port_path: string
50 :param api_level: level 1 is that the api only espablishes the connection to the uC and the "infinite" write and read buffers, you need to construct the instruction packages your self,
51 level 2 it wraps the full representation of the uC interfaces in objects that are made availible as variables on this object, defaults to 2
52 :type api_level: int, optional
53 """
54 self.__experiment_state = []
55 self.__experiment_state_timestamp = []
56 self.__connection = serial.Serial(serial_port_path,115200, timeout= None) #its USB so the speed setting gets ignored and it runes at max speed
57 self.__read_buffer = Queue()
58 self.__write_buffer_timed = Queue()
59 self.__write_buffer = Queue()
60 self.__communication_thread = threading.Thread(target=self.__thread_function)
61 self.__free_input_queue_spots_on_uc = -1
62 self.__last_timed_packet = 0
63 self.__api_level = api_level
64 if api_level == 2:
65 self.errors = []
66 self.spi = [Interface_SPI(self,0), Interface_SPI(self,1), Interface_SPI(self,2)]
67 self.i2c = [Interface_I2C(self,0), Interface_I2C(self,1), Interface_I2C(self,2)]
68 self.pin = []
69 for pin_id in range(55):
70 self.pin.append(Interface_PIN(self,pin_id))
71 self.async_to_chip = []
72 for async_id in range(8):
73 self.async_to_chip.append(Interface_Async(self,async_id,"TO_CHIP"))
74 self.async_from_chip = []
75 for async_id in range(8):
76 self.async_from_chip.append(Interface_Async(self,async_id,"FROM_CHIP"))
77 self.__communication_thread.start()

79 def update_state(self):
80 """update_state This method processes all availible messages from the uC and updates the internal representaion
81 in detail it distributes the recorded packages to the coresponding interfaces for processing.
82
83 level 2 only
84 """
85 while not self.__read_buffer.empty():
86 packet_to_process = self.__read_buffer.get()
87 if isinstance(packet_to_process, ErrorPacket):
88 header_for_sorting = packet_to_process.original_header()
89 else:
90 header_for_sorting = packet_to_process.header()
91 no_match = True
92 if header_for_sorting == Data32bitHeader.IN_SET_TIME:
93 self.__experiment_state.append(packet_to_process.value())
94 self.__experiment_state_timestamp.append(packet_to_process.time())
95 no_match = False
96 continue
97
98 for interface in self.async_to_chip + self.async_from_chip + self.spi + self.i2c:
99 if header_for_sorting in interface.header():
100 interface.process_packet(packet_to_process)
101 self.__read_buffer.task_done()
102 no_match = False
103 break
104 if header_for_sorting == self.pin[0].header()[0]:
105 self.pin[packet_to_process.value()].process_packet(packet_to_process)
106 self.__read_buffer.task_done()
107 no_match = False
108 continue
109 elif header_for_sorting in self.pin[0].header():
110 self.pin[packet_to_process.pin_id()].process_packet(packet_to_process)
111 self.__read_buffer.task_done()
112 no_match = False
113 continue
114 if no_match:
115 self.errors.append(str(packet_to_process))
116 self.__read_buffer.task_done()

118 def start_experiment(self):
119 """start_experiment This will reset the uC clock, enable that data is collected and that timed instructions are executed by the uC
120 latest after 72min stop_experiment has to be called, after which a new experiment can be programmed and the function can be called again.
121 """
122 packet_to_send = Data32bitPacket(header=Data32bitHeader.IN_SET_TIME,value=1)
123 self.send_packet(packet_to_send)
124
125 def stop_experiment(self, time = 0):
126 """stop_experiment this will stop recording and flush all not jet excecuted timed instructions
127
128 :param time: the time in us after start_experiment when this function should be called, defaults to 0 (execute instantly)
129 :type time: int, optional
130 """
131 packet_to_send = Data32bitPacket(header=Data32bitHeader.IN_SET_TIME,value=0,time=time)
132 self.send_packet(packet_to_send)

134 def experiment_state(self):
135 """experiment_state returns the state history
136
137 1 or larger: experiment is running
138 0 experiment stopped
139 -1 uC reset
140
141 :return: 2 lists: first the state second the uC timesteps in us of the state change
142 :rtype: ([int],[int])
143 """
144 return (self.__experiment_state, self.__experiment_state_timestamp)

146 def send_packet(self, packet_to_send):
147 """send_packet send a packet to the uC via the "infinite" buffer
148 needs a package object see package.py
149
150 :param packet_to_send: the package to be send
151 :type packet_to_send: Packet, or any subclass
152 """
153 if packet_to_send.header() == Data32bitHeader.IN_SET_TIME:
154 self.__last_timed_packet = packet_to_send.value()
155
156 if packet_to_send.time() == 0:
157 self.__write_buffer.put(packet_to_send)
158 else:
159 self.__write_buffer_timed.put(packet_to_send)
160 if packet_to_send.time() < self.__last_timed_packet:
161 logging.warning("the instructions are not sorted in time - execution order will be inconsistent")
162
163 def read_packet(self):
164 """read_packet returns one package from the uC via the "infinte" buffer
165
166 if there is no packet availible it blocks and waits until a packet becomes availible
167
168 :return: one package from the uC
169 :rtype: Packet, or any subclass
170 """
171 if self.__api_level == 1:
172 read_packet = self.__read_buffer.get()
173 self.__read_buffer.task_done()
174 return read_packet
175 else:
176 logging.error("reading raw packets is only availible in API level 1")

191 def close_connection(self):
192 """close_connection closes the serial connection to the uC and blocks until this is done
193 resets the uC too
194 """
195 self.__write_buffer.put(Data32bitPacket(Data32bitHeader.UC_CLOSE_CONNECTION))

197 self.__experiment_state_timestamp.append(-1)
198 self.__communication_thread.join()

203 self.__write_buffer.put(Data32bitPacket(Data32bitHeader.IN_RESET))

208 def __thread_function(self):
209 """__thread_function internal function managing the actual async communication with the uC in the background
210 """
211 idle_write_pc = False
212 idle_write_uc = 0
213 idle_read = False
214 while True:
215 if not self.__write_buffer_timed.empty() or not self.__write_buffer.empty():
216 idle_write_pc = False
217 if not self.__write_buffer.empty():
218 data_packet = self.__write_buffer.get()
219 if data_packet.header() == Data32bitHeader.UC_CLOSE_CONNECTION:
220 self.__connection.write(Data32bitPacket(Data32bitHeader.IN_RESET).to_bytearray())
221 self.__connection.close()
222 return
223 self.__connection.write(data_packet.to_bytearray())
224 logging.debug("send instant: "+str(data_packet))
225 self.__write_buffer.task_done()
226 else:
227 if self.__free_input_queue_spots_on_uc > 0 :
228 data_packet = self.__write_buffer_timed.get()
229 self.__free_input_queue_spots_on_uc -= 1
230 self.__connection.write(data_packet.to_bytearray())
231 logging.debug("send timed: "+str(data_packet))
232 self.__write_buffer_timed.task_done()
233 else:
234 idle_write_uc += 1
235 if idle_write_uc%2000 == 1:
236 packet_to_send = Data32bitPacket(Data32bitHeader.IN_FREE_INSTRUCTION_SPOTS)
237 self.__connection.write(packet_to_send.to_bytearray())
238 logging.debug("send timed: "+str(packet_to_send))
239 else:
240 idle_write_pc = True
241
242 if self.__connection.in_waiting >= 9:
243 idle_read = False
244 byte_packet = self.__connection.read(size = 9)
245 read_packet = Packet.from_bytearray(byte_packet)
246 logging.debug("read: "+str(read_packet))
247 if read_packet.header() is Data32bitHeader.OUT_FREE_INSTRUCTION_SPOTS:
248 self.__free_input_queue_spots_on_uc = read_packet.value()
249 idle_write_uc = 0
250 else:
251 self.__read_buffer.put(read_packet)
252 else:
253 idle_read = True
254
255 if idle_read and (idle_write_pc or idle_write_uc > 0):
256 sleep(0.000003)
257 idle_read = False
258 idle_write_pc = False
