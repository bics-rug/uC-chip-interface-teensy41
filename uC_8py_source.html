39class uC_api:
40 """
41 the class uC_api exposes the full interface to the uC as an object,
42 all the interface like i2c, spi, pin and async can be found as exposed variables
43
44 Interfaces need to be activated before use, and the activation is ackloaged by the uC
45 you might need to time.sleep(0.01) after the activation for the python object to update register the state change
46
47 if you want to record data you need to use function start_experiment() to be abel to see the data send back
48 the maximum time of each experiment is 72min, please call stop experiment before that to start agian
49
50 after you are done call close_connection to sever the serial connection to the uC,
51 the recorded data in the python object remains and can be processed after
52 """
53 def __init__(self, serial_port_path, api_level=2):
54 """__init__ creates the uC interface object and establishes the connection to the uC on the given port
55
56 :param serial_port_path: the path of your system to the serial port, eg. on linux it might be /dev/ttyAMC0 or higher, on mac /dev/tty.usbmodem<XXXXX> on windows <COM port>
57 :type serial_port_path: string
58 :param api_level: level 1 is that the api only espablishes the connection to the uC and the "infinite" write and read buffers, you need to construct the instruction packages your self,
59 level 2 it wraps the full representation of the uC interfaces in objects that are made availible as variables on this object, defaults to 2
60 :type api_level: int, optional
61 """
62 self.__experiment_state = []
63 self.__experiment_state_timestamp = []
64 connection = None
65 self.__read_buffer = Queue()
66 self.__write_buffer_timed = Queue()
67 self.__write_buffer = Queue()
68 self.__communication_thread = threading.Thread(target=self.__thread_function, args=(serial_port_path,))
69 self.__last_timed_packet = 0
70 self.__api_level = api_level
71 if api_level == 2:
72 # create all the interface objects
73 self.errors = []
74 self.spi = [Interface_SPI(self,0), Interface_SPI(self,1), Interface_SPI(self,2)]
75 self.i2c = [Interface_I2C(self,0), Interface_I2C(self,1), Interface_I2C(self,2)]
76 self.pin = []
77 for pin_id in range(55):
78 self.pin.append(Interface_PIN(self,pin_id))
79 self.async_to_chip = []
80 for async_id in range(8):
81 self.async_to_chip.append(Interface_Async(self,async_id,"TO_CHIP"))
82 self.async_from_chip = []
83 for async_id in range(8):
84 self.async_from_chip.append(Interface_Async(self,async_id,"FROM_CHIP"))
85 self.__communication_thread.start()
86 # wait for 10 seconds for the uC to align
87 connection = False

90 def update_state(self):
91 """update_state This method processes all availible messages from the uC and updates the internal representaion
92 in detail it distributes the recorded packages to the coresponding interfaces for processing.
93
94 level 2 only
95 """
96 # process all availible messages from the uC one by one
97 while not self.__read_buffer.empty():
98 # get the next package
99 packet_to_process = self.__read_buffer.get()
100 # choose the interface to process the package
101 if isinstance(packet_to_process, ErrorPacket):
102 header_for_sorting = packet_to_process.original_header()
103 else:
104 header_for_sorting = packet_to_process.header()
105
106 no_match = True
107 # high level experiment control packet
108 if header_for_sorting == Data32bitHeader.IN_SET_TIME:
109 self.__experiment_state.append(packet_to_process.value())
110 self.__experiment_state_timestamp.append(packet_to_process.time())
111 no_match = False
112 continue
113 # iterate though all interfaces and check if they signal they are responcible for that header
114 for interface in self.async_to_chip + self.async_from_chip + self.spi + self.i2c:
115 if header_for_sorting in interface.header():
116 interface.process_packet(packet_to_process)
117 self.__read_buffer.task_done()
118 no_match = False
119 break
120 # pins use all the same header, so we assing the package to the pin object with the same id
121 if header_for_sorting == self.pin[0].header()[0]:
122 self.pin[packet_to_process.value()].process_packet(packet_to_process)
123 self.__read_buffer.task_done()
124 no_match = False
125 continue
126 elif header_for_sorting in self.pin[0].header():
127 self.pin[packet_to_process.pin_id()].process_packet(packet_to_process)
128 self.__read_buffer.task_done()
129 no_match = False
130 continue
131 # if no interface is responcible for the package, we add it to the error package list
132 if no_match:
133 self.errors.append(str(packet_to_process))
134 self.__read_buffer.task_done()

174 def send_packet(self, packet_to_send):
175 """send_packet send a packet to the uC via the "infinite" buffer
176 needs a package object see package.py
177
178 :param packet_to_send: the package to be send
179 :type packet_to_send: Packet, or any subclass
180 """
181 # reset the time reference for the timed instructions
182 if packet_to_send.header() == Data32bitHeader.IN_SET_TIME:
183 self.__last_timed_packet = packet_to_send.value()
184 # put the packet in the buffer depending if it s instant or timed
185 if packet_to_send.time() == 0:
186 self.__write_buffer.put(packet_to_send)
187 else:
188 self.__write_buffer_timed.put(packet_to_send)
189 # check if the timed instructions are sorted in time
190 if packet_to_send.time() < self.__last_timed_packet:
191 logging.warning("the instructions are not sorted in time - execution order will be inconsistent")

221 def close_connection(self):
222 """close_connection closes the serial connection to the uC and blocks until this is done
223 resets the uC too
224 """
225 # place close connection packet in the write buffer, so the worker thread closes the connection and stop itself
226 self.__write_buffer.put(Data32bitPacket(Data32bitHeader.UC_CLOSE_CONNECTION))
227 # add reset to the experiment state history
228 self.__experiment_state.append(-1)
229 self.__experiment_state_timestamp.append(-1)
230 # wait for the worker thread to close the connection
231 self.__communication_thread.join()

242 def __check_first_connection(self,connection):
243 """__check_first_connection checks if the uC is responding and prints the firmware version
244 if the firmware version does not match the API version it will print a warning
245 """
246 connection_state = False
247 logging.info("send: opening connection - aligning commuication")
248 # write 9 bytes to the uC to align the communication
249 connection.write(ALIGN_BYTEARRAY)
250 # wait for 10 seconds for the uC to align
251 for i in range(40):
252 # check if the uC has send a packet
253 if connection.in_waiting >= 9:
254 byte_packet = connection.read(size = 9)
255 # do reading alingnment
256 byte_packet = byte_packet.lstrip(b'\xff')
257 sleep(0.05)
258 # if the packet is not complete, try to recover by reading the rest of the packet
259 if len(byte_packet) < 9:
260 if len(byte_packet) == 0:
261 # no packet (was only alignment bytes), jump to next iteration
262 continue
263 elif connection.in_waiting < 9-len(byte_packet):
264 # something went wrong, try again
265 logging.error("partial packet received, but not enough bytes send by uC, trying to recover by realigning")
266 connection.write(ALIGN_BYTEARRAY)
267 continue
268 # complete partial packet
269 else:
270 byte_packet = bytearray(byte_packet).extend(bytearray(connection.read(size = 9-len(byte_packet))))
271 # convert the byte packet to a packet object
272 read_packet = Packet.from_bytearray(byte_packet)
273 # check if the packet is the expected Success packet
274 if read_packet.header() == ErrorHeader.OUT_ALIGN_SUCCESS_VERSION:
275 logging.info("uC is ready - firmware version: "+str(read_packet.original_header())+"."+str(read_packet.original_sub_header())+"."+str(read_packet.value()))
276 # check if the firmware version matches the API version
277 if read_packet.original_header() != FIRMWARE_VERSION.FIRMWARE_VERSION_MAJOR or read_packet.original_sub_header() != FIRMWARE_VERSION.FIRMWARE_VERSION_MINOR or read_packet.value() < FIRMWARE_VERSION.FIRMWARE_VERSION_PATCH:
278 logging.warning("uC firmware version does not match the API version: \nfirmware version: "+str(read_packet.original_header())+"."+str(read_packet.original_sub_header())+"."+str(read_packet.value())+" \nAPI version: "+str(int(FIRMWARE_VERSION.FIRMWARE_VERSION_MAJOR))+"."+str(int(FIRMWARE_VERSION.FIRMWARE_VERSION_MINOR))+"."+str(int(FIRMWARE_VERSION.FIRMWARE_VERSION_PATCH)))
279 # connection is established
280 connection = True
281 connection_state = True
282 return True
283 else:
284 logging.warning("unknown packet received, while connecting to uC for the first time: "+str(read_packet))
285 sleep(0.25)
286 # connection failed after 40 tries/10sec
287 if connection_state == False:
288 logging.error("uC is not responding for 10 sec, wrong port?, no permission?")
289 return False

292 def __thread_function(self,serial_port_path):
293 """__thread_function internal function managing the actual async communication with the uC in the background
294 """
295 idle_write_pc = False
296 idle_write_uc = 0
297 idle_read = False
298 free_input_queue_spots_on_uc = -1
299 last_sent_time = 0
300 last_free_spots = 0
301 packet_send = 0
302 exec_running = 0
303 request_free_input_queue_spots = False
304 connection = serial.Serial(serial_port_path,115200, timeout= None, write_timeout=0) #its USB so the speed setting gets ignored and it runes at max speed
305 # init communication by forcing the uC to align
306 if not self.__check_first_connection(connection):
307 connection.close()
308 return
309
310 # start communication
311 while True:
312 # check if there is something to send
313 if not self.__write_buffer_timed.empty() or not self.__write_buffer.empty():
314 # set loop slowdown condition flags to false
315 idle_write_pc = False
316 # first write the instant packets
317 if not self.__write_buffer.empty():
318 data_packet = self.__write_buffer.get()
319 # check and close the connection if requested by API
320 if data_packet.header() == Data32bitHeader.UC_CLOSE_CONNECTION:
321 connection.write(Data32bitPacket(Data32bitHeader.IN_RESET).to_bytearray())
322 connection.close()
323 return
324 # else send the packet
325 connection.write(data_packet.to_bytearray())
326 logging.debug("send instant: "+str(data_packet))
327 self.__write_buffer.task_done()
328 # then write the timed packets
329 else:
330 # check if there is space in the uC input queue
331 if free_input_queue_spots_on_uc > 0 :
332 # send the packet and decrease the free input queue spots reference in the API
333 data_packet = self.__write_buffer_timed.get()
334 free_input_queue_spots_on_uc -= 1
335 connection.write(data_packet.to_bytearray())
336 last_sent_time = data_packet.time()
337 packet_send += 1
338 logging.debug("send timed: "+str(data_packet))
339 self.__write_buffer_timed.task_done()
340 else:
341 # request the free input queue spots from the uC,
342 # first request is send instantly, then every 200th loop run through
343 # to not overload the uC with requests, uC will also report the free input
344 # queue spots when it frees up space and the queue was full before
345 idle_write_uc += 1
346 if idle_write_uc%200 == 1:
347 if request_free_input_queue_spots == False:
348 request_free_input_queue_spots = True
349 packet_to_send = Data32bitPacket(Data32bitHeader.IN_FREE_INSTRUCTION_SPOTS)
350 connection.write(packet_to_send.to_bytearray())
351 logging.debug("send request: "+str(packet_to_send))
352 else:
353 # set write loop slowdown condition flag
354 idle_write_pc = True
355
356 # check if there is a package to read from the serial connection,
357 # read 20 packages under the assumption that the pc is faster and
358 # the uC will send more packages
359 # stops if there is nothing to read
360 for i in range(20):
361 if connection.in_waiting >= 9:
362 idle_read = False
363 byte_packet = connection.read(size = 9)
364 # remove alignment bytes
365 byte_packet = byte_packet.lstrip(b'\xff')
366 # check if the packet is complete
367 if len(byte_packet) < 9:
368 if len(byte_packet) != 0:
369 #initiate alignment for incompleate packets
370 logging.warning("outgoing uC alignment needed, shifted by "+str(len(byte_packet))+ " bytes")
371 # partial packet not recoverable
372 if connection.in_waiting < 9-len(byte_packet):
373 logging.error("partial packet received, but not enough bytes send by uC, trying to recover by realigning")
374 connection.write(ALIGN_BYTEARRAY)
375 continue
376 # to complete partial packet
377 byte_packet = bytearray(byte_packet).extend(bytearray(connection.read(size = 9-len(byte_packet))))
378
379 else:
380 logging.debug("alignment sucesss - no incoming alignment error")
381 # no packet, jump to next iteration
382 continue
383 # is now aligned
384 try:
385 # convert the byte packet to a packet object
386 read_packet = Packet.from_bytearray(byte_packet)
387 except:
388 # packet was malformed, force alignment sequence
389 logging.error("packet is malformed, maybe misaligned, trying to recover by realigning")
390 connection.write(ALIGN_BYTEARRAY)
391 continue
392 # packet is complete and valid
393 logging.debug("read: "+str(read_packet))
394 # catch the special case of the uC reporting free input queue spots
395 if read_packet.header() is Data32bitHeader.OUT_FREE_INSTRUCTION_SPOTS:
396 # save the free input queue spots in the API
397 if read_packet.time() > last_sent_time and exec_running > 0:
398 logging.warning("Timing exec squewed, increase buffer size in firmware, last sent time: "+\
399 str(last_sent_time)+" uC time: "+str(read_packet.time())+\
400 "\npackets send: "+str(packet_send)+" for free spots: "+str(last_free_spots))
401 else:
402 logging.debug("uC reports "+str(read_packet.value())+" free input queue spots at time "+\
403 str(read_packet.time())+"\nwaiting on PC: "+str(self.__write_buffer_timed.qsize())+\
404 " with time starting from: "+str(last_sent_time)+\
405 "\npackets send: "+str(packet_send)+" for free spots: "+str(last_free_spots))
406 last_free_spots = read_packet.value()
407 packet_send = 0
408 # set one less then availible, because of bug the pc will send to much
409 free_input_queue_spots_on_uc = read_packet.value()
410 if free_input_queue_spots_on_uc == 0:
411 idle_write_uc = 2
412 else:
413 idle_write_uc = 0
414 request_free_input_queue_spots = False
415 # catch the special case of the uC reporting an malformed packet from the API
416 elif read_packet.header() is ErrorHeader.OUT_ERROR_UNKNOWN_INSTRUCTION or read_packet.header() is ErrorHeader.OUT_ERROR_UNKNOWN_CONFIGURATION:
417 logging.error("uC is reporting that it cant understand a send packet, either API and firmware are a different version or communication is not aligned, trying to recover by realigning")
418 connection.write(ALIGN_BYTEARRAY)
419 # keep track of the experiment state, so we know when to issue a warning for execution time squew
420 elif read_packet.header() == Data32bitHeader.IN_SET_TIME:
421 exec_running = read_packet.value()
422 logging.info("Experiment state changed to: "+str(exec_running))
423 self.__read_buffer.put(read_packet)
424 # normal packet, send to the read buffer for further processing by the main thread
425 else:
426 self.__read_buffer.put(read_packet)
427 else:
428 # set read loop slowdown condition flag, as there is nothing to read
429 idle_read = True
430 break
431
432 # slow down the loop if there is nothing to do
433 if idle_read and (idle_write_pc or idle_write_uc > 0):
434 sleep(0.000003)
435 idle_read = False
436 idle_write_pc = False
