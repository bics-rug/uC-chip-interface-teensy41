250 def process_packet(self, packet):
251 """ process a packet from the uC, update the status of the interface object and store the data
252 @param packet: packet object to be processed
253 """
254 if packet.header() in self.__header:
255 # process any acknolage config packet (config sucessfully set on the uC)
256 if packet.header() == self.__header[0]:
257 # interface is active
258 if packet.config_header() == ConfigSubHeader.CONF_ACTIVE:
259 self.__status = 2
260 self.__status_timestamp = packet.time()
261 return
262 # the mode of the interface
263 elif packet.config_header() == ConfigSubHeader.CONF_TYPE:
264 """@todo replace with type"""
265 if packet.value() == 0:
266 self.__mode = "4Phase_Chigh_Dhigh"
267 elif packet.value() == 1:
268 self.__mode = "4Phase_Clow_Dhigh"
269 elif packet.value() == 10:
270 self.__mode = "2Phase"
271 elif packet.value() == 20:
272 self.__mode = "4Phase_MCP23017"
273 else:
274 logging.error("unknown async type returned")
275 self.__mode_timestamp = packet.time()
276 return
277 # the ack pin of the interface
278 elif packet.config_header() == ConfigSubHeader.CONF_ACK:
279 self.__ack_pin = packet.value()
280 self.__ack_pin_timestamp = packet.time()
281 return
282 # the request pin of the interface
283 elif packet.config_header() == ConfigSubHeader.CONF_REQ:
284 self.__req_pin = packet.value()
285 self.__req_pin_timestamp = packet.time()
286 return
287 # the data width of the interface
288 elif packet.config_header() == ConfigSubHeader.CONF_WIDTH:
289 self.__data_width = packet.value()
290 self.__data_width_timestamp = packet.time()
291 return
292 # the data pins of the interface
293 elif packet.config_header() < 32:
294 self.__data_pins[packet.config_header()] = packet.value()
295 self.__data_pins_timestamp[packet.config_header()] = packet.time()
296 return
297 # the request line delay of the interface
298 elif packet.config_header() == ConfigSubHeader.CONF_REQ_DELAY:
299 self.__req_delay = packet.value()
300 self.__req_delay_timestamp = packet.time()
301 return
302 # process any data packet (data send from the uC to the chip or recived from the chip)
303 elif packet.header() == self.__header[1]:
304 if self.__direction == "TO_CHIP":
305 self.__data_to_chip.append(packet.value())
306 self.__data_to_chip_times.append(packet.time())
307 return
308 elif self.__direction == "FROM_CHIP":
309 self.__data_from_chip.append(packet.value())
310 self.__data_from_chip_times.append(packet.time())
311 return
312 # all other packets are errors or not supported, disable the interface
313 self.__errors.append(str(packet))
314 self.__status = -1

316 def activate(self, req_pin, ack_pin, data_width, data_pins, mode="4Phase_Chigh_Dhigh", req_delay = 0, time = 0):
317 """ activate the interface on the uC with the given parameters
318
319 Mode:
320 - "4Phase_Chigh_Dhigh" means 4 phase clock with high active clock and high active data
321 - "4Phase_Clow_Dhigh" means 4 phase clock with low active clock and high active data
322 - "2Phase" means 2 phase clock with high active data
323 - "4Phase_MCP23017" means 4 phase clock with high active clock and high active data and MCP23017 port extender, pin IDs are not used in this mode
324
325 @param req_pin: request pin id on the uC
326 @param ack_pin: ack pin id on the uC
327 @param data_width: data width of the interface
328 @param data_pins: list of data pin ids on the uC
329 @param mode: mode of the interface - "4Phase_Chigh_Dhigh", "4Phase_Clow_Dhigh", "2Phase", "4Phase_MCP23017"
330 @param req_delay: how much the request of the handshake is delayed multiplied by 20us
331 @param time: the exec_time when the uc should activate the interface, 0 means as soon as possible
332 """
333 # only activate if not already activated or pending activation
334 if self.__status >= 1:
335 logging.warning("ASYNC_to_chip interface "+str(self.__header[0])+" is already activated or waiting activation, doing nothing")
336 else:
337 # send the configuration to the uC as individual packets
338 if mode == "4Phase_Chigh_Dhigh":
339 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_TYPE, value = 0, time = time))
340 self.__status = 1
341 elif mode == "4Phase_Clow_Dhigh":
342 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_TYPE, value = 1, time = time))
343 self.__status = 1
344 elif mode == "2Phase_Chigh_Dhigh":
345 logging.warning("pin mode not implmented yet")
346 return
347 elif mode == "2Phase_Clow_Dhigh":
348 logging.warning("pin mode not implmented yet")
349 return
350 elif mode == "4Phase_MCP23017":
351 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_TYPE, value = 20, time = time))
352 self.__status = 1
353 else:
354 logging.error("pin.activate got wrong type "+str(pin_mode)+" only 4Phase_Chigh_Dhigh, 4Phase_Clow_Dhigh, 2Phase_Chigh_Dhigh, 2Phase_Clow_Dhigh are allowed, more modes implemted on request")

356 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_ACK, value = ack_pin, time = time))
357 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_REQ, value = req_pin, time = time))
358 self.__api.send_packet(ConfigPacket(header = self.__header[0], config_header = ConfigSubHeader.CONF_WIDTH, value = data_width, time = time))
359 for pin in range(data_width):

371 """
372 self.update()
