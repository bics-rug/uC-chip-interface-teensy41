32/*
33 This programm is supposed to be used for interfacing with small Async or Neuromorphic chips,
34 the pin and interface configuration is uploaded on runtime via configuration packets,
35 so for most chips/modules a change of the uC firmware is not nessesary.
36
37 the Idea behind the firmware is that you can send command packets with a exec time in the future,
38and the command is executed at that time via interrupt. (see instruction_exec.h/cpp)
39 if you send exec_time=0 the execution is done right away. The exec_time needs to be sorted and assending.
40 the execuion is done via interrupts the execution only happens every ~100us => see instruction_exec.h
41 timesteps are in usec and the role over occurs after ~71min after boot of the uC. roleover is not supported.
42
43 the folloeing interfaces are supported:
44 - SPI - Serial Periferal Interface for memory and fifo access (see "Interface_spi.h/cpp")
45 - AER - paralell databus with 4-phase handshake (see "AER_tochip.h/cpp" and "AER_from_chip.h/cpp")
46 - PIN - manual pin raising and lowering (see "Interface_pin.h/cpp")
47
48
49 see datatypes.h for all available commands and thier use.
50
51 to start recording and execution set the TIME to anything other than 0, to stop execution and recording sent TIME to 0.
52
53 recording of events is handeled in "isr_helper.h/cpp"
54
55 the coding style is following the Arduino Style of having interfaces staticly allocatated for access.
56 In a future version this shoud be at least changed to static Object Oriented Programing
57
58*/

74/*
75 the main loop just handels comunication with the host via the serial interface and instuction execution if exec_time == 0,
76 in all other cases the instruction is stored in the instruction ring buffer.
77*/
78void loop() {
79 if (Serial.available() >= sizeof(packet_t)) {
80
81 // read instruction packet
82 packet_t current_instruction;
83 // because we use different storage allignment we need to transfer each byte individual
84 uint8_t position;
85 for (position = 0; position < sizeof(packet_t); position++){
86 Serial.readBytes(&(current_instruction.bytes[position]), 1);
87 }
88
89 // exec instruction if exec_time == 0
90 if (current_instruction.data.exec_time == 0) {
91 exec_instruction(&current_instruction, false);
92 }
93
94 // if instruction buffer is not full store instruction
95 else if (input_ring_buffer_start != (input_ring_buffer_next_free + 1) % INPUT_BUFFER_SIZE) {
96 noInterrupts();
97 copy_packet(&current_instruction,&input_ring_buffer[input_ring_buffer_next_free]);
98 //input_ring_buffer[input_ring_buffer_next_free] = current_instruction;
99 input_ring_buffer_next_free = (input_ring_buffer_next_free + 1) % INPUT_BUFFER_SIZE;
100 interrupts();
101 }
102
103 // if instruction buffer is full send error, error_message not used because send without que
104 else {
105 error_message_bypass_buffer(OUT_ERROR_INPUT_FULL,current_instruction.data.header,current_instruction.data.value);
106 }
107 }
108
109 // write one element of the output buffer
110 if (output_buffer_read && Serial.availableForWrite()){
111 send_output_ring_buffer_first();
112 }
113
114 // send error if output buffer is full, DoS
115 //else if (output_ring_buffer_start == (output_ring_buffer_next_free + 1) % OUTPUT_BUFFER_SIZE) {
116 // error_message_bypass_buffer(OUT_ERROR,OUT_ERROR_OUTPUT_FULL,output_ring_buffer_next_free);
117 //}
118
119 if (loop_runs_without_gpio_interrups > 0){
120 loop_runs_without_gpio_interrups--;
121 if (loop_runs_without_gpio_interrups == 0){
122 enable_gpio_interrupt();
123 }
124 }
125}
