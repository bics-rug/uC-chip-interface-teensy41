72bool is_output_buffer_not_full(){
73 uint32_t free_spots = 0;
74 if (output_ring_buffer_start > output_ring_buffer_next_free){
75 free_spots = output_ring_buffer_start-output_ring_buffer_next_free-1;
76 }
77 else{
78 free_spots = OUTPUT_BUFFER_SIZE-output_ring_buffer_next_free+output_ring_buffer_start-1;
79 }
80 // do most likey case first to speed up the request.
81 if (free_spots > 5 ) return true;
82 else if (free_spots > 1) {
83 loop_runs_without_gpio_interrups +=10;
84 disable_gpio_interrupt();
85 if (is_realtime){
86 noInterrupts();
87 output_ring_buffer[output_ring_buffer_next_free].error.header = OUT_WARNING_DATA_COLLECTION_SQUEUED;
88 output_ring_buffer[output_ring_buffer_next_free].error.org_header = OUT_WARNING_DATA_COLLECTION_SQUEUED;
89 output_ring_buffer[output_ring_buffer_next_free].error.value = free_spots;
90 output_ring_buffer_next_free = (output_ring_buffer_next_free + 1) % OUTPUT_BUFFER_SIZE;
91 is_realtime = false;
92 interrupts();
93 }
94 return true;
95 }
96 else if (free_spots == 1) {
97 noInterrupts();
98 output_ring_buffer[output_ring_buffer_next_free].error.header = OUT_ERROR_OUTPUT_FULL;
99 output_ring_buffer[output_ring_buffer_next_free].error.org_header = OUT_ERROR_OUTPUT_FULL;
100 output_ring_buffer[output_ring_buffer_next_free].error.value = 1;
101 output_ring_buffer_next_free = (output_ring_buffer_next_free + 1) % OUTPUT_BUFFER_SIZE;
102 loop_runs_without_gpio_interrups +=5;
103 disable_gpio_interrupt();
104 interrupts();
105 return false;
106 }
107 else {
108 noInterrupts();
109 uint16_t index = output_ring_buffer_next_free == 0 ? OUTPUT_BUFFER_SIZE-1 : output_ring_buffer_next_free-1;
110 output_ring_buffer[index].error.header = OUT_ERROR_OUTPUT_FULL;
111 output_ring_buffer[index].error.org_header = OUT_ERROR_OUTPUT_FULL;
112 output_ring_buffer[index].error.value++;
113 loop_runs_without_gpio_interrups +=5;
114 disable_gpio_interrupt();
115 interrupts();
116 return false;
117 }
118}

134void send_input_ring_buffer_free_spots(){
135 packet_t data_packet;
136 data_packet.data.header = OUT_FREE_INSTRUCTION_SPOTS;
137 data_packet.data.exec_time = micros()-offset_time;
138 uint32_t free_spots = 0;
139 noInterrupts();
140 if (input_ring_buffer_start > input_ring_buffer_next_free){
141 free_spots = input_ring_buffer_start-input_ring_buffer_next_free-1;
142 }
143 else{
144 free_spots = INPUT_BUFFER_SIZE-input_ring_buffer_next_free+input_ring_buffer_start-1;
145 }
146 interrupts();
147 data_packet.data.value = free_spots;
148 uint8_t position;
149 // due to different storage alignment need to write data bytes individual
150 for (position = 0; position < sizeof(packet_t); position++) Serial.write(data_packet.bytes[position]);
151 //send confirmation
152 send_data32(IN_FREE_INSTRUCTION_SPOTS,0,true);
153

156void add_input_packet(packet_t* input_packet){
157 if (is_input_buffer_not_full()){
158 noInterrupts();
159 copy_packet(input_packet,&input_ring_buffer[input_ring_buffer_next_free]);
160 input_ring_buffer_next_free = (input_ring_buffer_next_free + 1) % INPUT_BUFFER_SIZE;
161 interrupts();
162 }
163 // if instruction buffer is full send error, error_message not used because send without que
164 else {
165 error_message_bypass_buffer(OUT_ERROR_INPUT_FULL,input_packet->data.header,input_packet->data.value);
166 }

169void send_output_ring_buffer(){
170 noInterrupts();
171 if (!read_active){
172 read_active = true;
173 interrupts();
174 while (output_ring_buffer_start != output_ring_buffer_next_free){
175 noInterrupts();
176 packet_t out = copy_packet(&output_ring_buffer[output_ring_buffer_start]);
177 output_ring_buffer_start = (output_ring_buffer_start + 1) % OUTPUT_BUFFER_SIZE;
178 interrupts();
179 uint8_t position;
180 // due to different storage alignment need to write data bytes individual
181 for (position = 0; position < sizeof(packet_t); position++) Serial.write(out.bytes[position]);
182 }
183 read_active = false;
184 //send confirmation
185 send_data32(IN_READ,0,true);
186 send_output_ring_buffer_first();
187 }
188}
