72bool is_output_buffer_not_full(){
73 uint32_t free_spots = 0;
74 if (output_ring_buffer_start > output_ring_buffer_next_free){
75 free_spots = output_ring_buffer_start-output_ring_buffer_next_free-1;
76 }
77 else{
78 free_spots = OUTPUT_BUFFER_SIZE-output_ring_buffer_next_free+output_ring_buffer_start-1;
79 }
80 // do most likey case first to speed up the request.
81 if (free_spots > 5 ) return true;
82 else if (free_spots > 1) {
83 loop_runs_without_gpio_interrups +=10;
84 disable_gpio_interrupt();
85 if (is_realtime){
86 noInterrupts();
87 output_ring_buffer[output_ring_buffer_next_free].error.header = OUT_WARNING_DATA_COLLECTION_SQUEUED;
88 output_ring_buffer[output_ring_buffer_next_free].error.org_header = OUT_WARNING_DATA_COLLECTION_SQUEUED;
89 output_ring_buffer[output_ring_buffer_next_free].error.value = free_spots;
90 output_ring_buffer_next_free = (output_ring_buffer_next_free + 1) % OUTPUT_BUFFER_SIZE;
91 is_realtime = false;
92 interrupts();
93 }
94 return true;
95 }
96 else if (free_spots == 1) {
97 noInterrupts();
98 output_ring_buffer[output_ring_buffer_next_free].error.header = OUT_ERROR_OUTPUT_FULL;
99 output_ring_buffer[output_ring_buffer_next_free].error.org_header = OUT_ERROR_OUTPUT_FULL;
100 output_ring_buffer[output_ring_buffer_next_free].error.value = 1;
101 output_ring_buffer_next_free = (output_ring_buffer_next_free + 1) % OUTPUT_BUFFER_SIZE;
102 loop_runs_without_gpio_interrups +=5;
103 disable_gpio_interrupt();
104 interrupts();
105 return false;
106 }
107 else {
108 noInterrupts();
109 uint16_t index = output_ring_buffer_next_free == 0 ? OUTPUT_BUFFER_SIZE-1 : output_ring_buffer_next_free-1;
110 output_ring_buffer[index].error.value++;
111 loop_runs_without_gpio_interrups +=5;
112 disable_gpio_interrupt();
113 interrupts();
114 return false;
115 }
116}

118void send_input_ring_buffer_free_spots(){
119 packet_t data_packet;
120 data_packet.data.header = OUT_FREE_INSTRUCTION_SPOTS;
121 data_packet.data.exec_time = micros()-offset_time;
122 if (input_ring_buffer_start > input_ring_buffer_next_free){
123 data_packet.data.value = input_ring_buffer_start-input_ring_buffer_next_free-1;
124 }
125 else{
126 data_packet.data.value = INPUT_BUFFER_SIZE-input_ring_buffer_next_free+input_ring_buffer_start-1;
127 }
128 uint8_t position;
129 // due to different storage alignment need to write data bytes individual
130 for (position = 0; position < sizeof(packet_t); position++) Serial.write(data_packet.bytes[position]);
131 //send confirmation
132 send_data32(IN_FREE_INSTRUCTION_SPOTS,0,true);
133

136void send_output_ring_buffer(){
137 noInterrupts();
138 if (!read_active){
139 read_active = true;
140 interrupts();
141 while (output_ring_buffer_start != output_ring_buffer_next_free){
142 noInterrupts();
143 packet_t out = copy_packet(&output_ring_buffer[output_ring_buffer_start]);
144 output_ring_buffer_start = (output_ring_buffer_start + 1) % OUTPUT_BUFFER_SIZE;
145 interrupts();
146 uint8_t position;
147 // due to different storage alignment need to write data bytes individual
148 for (position = 0; position < sizeof(packet_t); position++) Serial.write(out.bytes[position]);
149 }
150 read_active = false;
151 //send confirmation
152 send_data32(IN_READ,0,true);
153 send_output_ring_buffer_first();
154 }
155}
